### 常用库 argparse 简介

```python
parser.add_argument(name or flags...[, action][, nargs][, const][, default][, type][,choices][, required][, help][, metavar][, dest])
```

- name or flags: 普通参数或flag参数选项参数的名称或标签，例如 epochs 或者 -e, --epochs。Flag参数不需要指定参数值，只需要带有参数名即可。
- action: 命令行遇到flags参数时的动作。有两个常见的动作，store_true：设定flag参数为true；store_false：设定flag参数为False。注意：如果直接运行程序，默认不读取该变量，要使用必须要进行传参，例如：python try.py --epochs
- nargs: 应该读取的命令行参数个数，可以是具体的数字，或者是?号，当不指定值时对于 Positional argument 使用 default，对于 Optional argument 使用 const；或者是 * 号，表示 0 或多个参数；或者是 + 号表示 1 或多个参数。
- default: 不指定参数时该参数的默认值。
- type: 命令行参数应该被转换成的数据类型。
- required: 是否为必选参数或可选参数。
- help: 参数的帮助信息。
- metavar： 在 usage 说明中的参数名称，对于必选参数，默认就是参数名称，对于可选参数默认是全大写的参数名称。
- dest： 解析后的参数名称，默认情况下，对于可选参数选取最长的名称，中划线转换为下划线.
- choices： 参数可允许的值的一个容器。
- const： action 和 nargs 所需要的常量值。
- store_const：表示赋值为const；
- append：将遇到的值存储成列表，也就是如果参数重复则会保存多个值;
- append_const：将参数规范中定义的一个值保存到一个列表；
- count：存储遇到的次数；此外，也可以继承 argparse.Action 自定义参数解析

### argparse 使用举例
```python
parser.add_argument('--epochs', type=int, default=30)
```

# Python 开发-测试一体化框架
## 概述
本文提供一个轻量级的**开发-测试一体化框架**，支持编写业务模块时自动生成对应的测试模板，并提供一键运行测试的能力，适配Python开发新手使用。

## 框架设计思路
1. **目录结构自动生成**：创建标准化的项目结构（开发目录 + 测试目录）
2. **模块生成器**：输入模块名即可生成业务代码模板和对应的测试代码模板
3. **测试运行器**：提供一键运行所有测试或指定测试的功能
4. **规范约束**：遵循Python测试最佳实践（unittest/pytest）

## 完整框架代码
```python
import os
import sys
import argparse
import shutil
from datetime import datetime
import unittest

# 框架核心配置
CONFIG = {
    "BASE_DIR": os.path.abspath("dev_test_project"),  # 项目根目录
    "SRC_DIR": "src",  # 开发代码目录
    "TEST_DIR": "tests",  # 测试代码目录
    "TEMPLATE_HEADER": f"""\"\"\"
Auto-generated by Dev-Test Framework
Created at: {datetime.now().strftime('%Y-%m-%d %H:%M:%S')}
Do not modify the header manually
\"\"\"
"""
}

class DevTestFramework:
    def __init__(self):
        """初始化框架，创建基础目录结构"""
        self.base_dir = CONFIG["BASE_DIR"]
        self.src_dir = os.path.join(self.base_dir, CONFIG["SRC_DIR"])
        self.test_dir = os.path.join(self.base_dir, CONFIG["TEST_DIR"])
        self._init_directories()

    def _init_directories(self):
        """初始化项目目录结构"""
        dirs = [self.base_dir, self.src_dir, self.test_dir]
        for dir_path in dirs:
            if not os.path.exists(dir_path):
                os.makedirs(dir_path)
                print(f"✅ 创建目录: {dir_path}")
            else:
                print(f"ℹ️  目录已存在: {dir_path}")
        
        # 创建__init__.py文件，使目录成为Python包
        for dir_path in [self.src_dir, self.test_dir]:
            init_file = os.path.join(dir_path, "__init__.py")
            if not os.path.exists(init_file):
                with open(init_file, "w", encoding="utf-8") as f:
                    f.write(CONFIG["TEMPLATE_HEADER"])

    def create_module(self, module_name):
        """
        创建业务模块和对应的测试模块
        :param module_name: 模块名称（如user、order、payment）
        """
        if not module_name.isidentifier():
            raise ValueError(f"❌ 模块名 '{module_name}' 不合法，必须是有效的Python标识符")
        
        # 1. 创建业务模块文件
        src_module_path = os.path.join(self.src_dir, f"{module_name}.py")
        if os.path.exists(src_module_path):
            print(f"⚠️  业务模块 {module_name}.py 已存在，跳过创建")
        else:
            with open(src_module_path, "w", encoding="utf-8") as f:
                f.write(CONFIG["TEMPLATE_HEADER"])
                f.write(f"""

class {module_name.capitalize()}Handler:
    \"\"\"{module_name}业务处理类\"\"\"
    def __init__(self):
        pass

    def process(self, data):
        \"\"\"核心业务处理方法
        :param data: 输入数据
        :return: 处理结果
        \"\"\"
        # TODO: 实现你的业务逻辑
        return {{
            "status": "success",
            "data": data,
            "module": "{module_name}"
        }}


def main():
    \"\"\"模块入口函数\"\"\"
    handler = {module_name.capitalize()}Handler()
    result = handler.process({{"test": "data"}})
    print(result)


if __name__ == "__main__":
    main()
""")
            print(f"✅ 生成业务模块: {src_module_path}")

        # 2. 创建测试模块文件
        test_module_path = os.path.join(self.test_dir, f"test_{module_name}.py")
        if os.path.exists(test_module_path):
            print(f"⚠️  测试模块 test_{module_name}.py 已存在，跳过创建")
        else:
            with open(test_module_path, "w", encoding="utf-8") as f:
                f.write(CONFIG["TEMPLATE_HEADER"])
                f.write(f"""
import unittest
import sys
sys.path.append("{self.base_dir}")
from {CONFIG["SRC_DIR"]}.{module_name} import {module_name.capitalize()}Handler


class Test{module_name.capitalize()}Handler(unittest.TestCase):
    \"\"\"{module_name}模块测试类\"\"\"
    
    def setUp(self):
        \"\"\"测试前置条件（每个测试方法执行前调用）\"\"\"
        self.handler = {module_name.capitalize()}Handler()

    def test_process_success(self):
        \"\"\"测试process方法正常场景\"\"\"
        test_data = {{"key": "value"}}
        result = self.handler.process(test_data)
        
        # 断言结果
        self.assertEqual(result["status"], "success")
        self.assertEqual(result["module"], "{module_name}")
        self.assertEqual(result["data"], test_data)

    def test_process_edge_case(self):
        \"\"\"测试process方法边界场景（TODO: 补充测试逻辑）\"\"\"
        # TODO: 实现边界条件测试
        pass


if __name__ == "__main__":
    unittest.main(verbosity=2)
""")
            print(f"✅ 生成测试模块: {test_module_path}")

    def run_tests(self, test_module=None):
        """
        运行测试
        :param test_module: 指定测试模块名（如user），None则运行所有测试
        """
        # 将项目根目录加入Python路径
        sys.path.insert(0, self.base_dir)
        
        if test_module:
            # 运行指定模块的测试
            test_file = os.path.join(self.test_dir, f"test_{test_module}.py")
            if not os.path.exists(test_file):
                print(f"❌ 测试模块 {test_file} 不存在")
                return
            
            # 加载并运行指定测试
            loader = unittest.TestLoader()
            suite = loader.discover(self.test_dir, pattern=f"test_{test_module}.py")
        else:
            # 运行所有测试
            suite = unittest.TestLoader().discover(self.test_dir, pattern="test_*.py")
        
        # 执行测试并输出结果
        runner = unittest.TextTestRunner(verbosity=2)
        result = runner.run(suite)
        
        # 输出测试总结
        print("\\n" + "="*50)
        print(f"测试完成 | 运行用例数: {result.testsRun}")
        print(f"失败用例数: {len(result.failures)} | 错误用例数: {len(result.errors)}")
        print("="*50)
        
        return result.wasSuccessful()

    def clean_project(self, confirm=True):
        """
        清理项目目录（谨慎使用）
        :param confirm: 是否需要确认
        """
        if confirm:
            answer = input(f"⚠️  确定要删除整个项目目录 {self.base_dir} 吗？(y/N): ")
            if answer.lower() != "y":
                print("取消清理操作")
                return
        
        if os.path.exists(self.base_dir):
            shutil.rmtree(self.base_dir)
            print(f"✅ 已删除项目目录: {self.base_dir}")
        else:
            print("❌ 项目目录不存在")

def main():
    """框架入口函数"""
    parser = argparse.ArgumentParser(description="Python开发-测试一体化框架")
    subparsers = parser.add_subparsers(dest="command", help="可用命令")

    # 创建模块命令
    create_parser = subparsers.add_parser("create", help="创建新模块")
    create_parser.add_argument("module", help="模块名称（如user、order）")

    # 运行测试命令
    test_parser = subparsers.add_parser("test", help="运行测试")
    test_parser.add_argument("-m", "--module", help="指定测试模块（可选，默认运行所有）")

    # 清理项目命令
    clean_parser = subparsers.add_parser("clean", help="清理项目目录")
    clean_parser.add_argument("-f", "--force", action="store_true", help="强制清理，无需确认")

    # 解析参数
    args = parser.parse_args()
    framework = DevTestFramework()

    # 执行对应命令
    if args.command == "create":
        framework.create_module(args.module)
    elif args.command == "test":
        framework.run_tests(args.module)
    elif args.command == "clean":
        framework.clean_project(not args.force)
    else:
        parser.print_help()

if __name__ == "__main__":
    main()
```

## 使用方法
### 1. 保存框架文件
将上述代码保存为 `dev_test_framework.py`

### 2. 基本操作命令
```bash
# 1. 创建新模块（例如创建user模块）
python dev_test_framework.py create user

# 2. 运行所有测试
python dev_test_framework.py test

# 3. 运行指定模块的测试（例如只运行user模块的测试）
python dev_test_framework.py test -m user

# 4. 清理项目目录（谨慎使用）
python dev_test_framework.py clean
# 强制清理（无需确认）
python dev_test_framework.py clean -f
```

### 3. 项目目录结构
运行 `create user` 后会生成如下结构：
```
dev_test_project/
├── src/
│   ├── __init__.py
│   └── user.py       # 业务模块
└── tests/
    ├── __init__.py
    └── test_user.py  # 测试模块
```

### 4. 开发流程示例
1. 运行 `python dev_test_framework.py create order` 创建订单模块
2. 编辑 `dev_test_project/src/order.py` 实现业务逻辑
3. 编辑 `dev_test_project/tests/test_order.py` 补充测试用例
4. 运行 `python dev_test_framework.py test -m order` 测试订单模块
5. 完成后运行 `python dev_test_framework.py test` 执行全量测试
